<script>
  (function() {
    // Konfigurasi delay dalam milidetik (misal 800ms = 0.8 detik)
    const DELAY_MS = 800;

    function playAudioWithDelay() {
      // Cari elemen audio di dalam container
      const audioElements = document.querySelectorAll('#audio-container audio');
      
      audioElements.forEach(audio => {
        // Matikan autoplay default agar tidak bentrok
        audio.autoplay = false;
        audio.pause(); 
        
        // Jalankan audio setelah delay
        setTimeout(() => {
          audio.currentTime = 0;
          audio.play().catch(e => console.log("Audio play blocked/failed:", e));
        }, DELAY_MS);
      });
    }

    // Eksekusi fungsi
    if (document.readyState === 'complete') {
      playAudioWithDelay();
    } else {
      window.addEventListener('load', playAudioWithDelay);
    }
  })();
</script>
